## Примеры тестов

```nova
// custom_iter.nv
type CountDown { current int }

impl Iter[int] for CountDown {
    fn next() -> Option[int] {
        if @current <= 0 {
            None
        } else {
            let v = @current
            @current = @current - 1
            Some(v)
        }
    }
}

test "CountDown next() sequence" {
    let mut cd = CountDown { current: 3 }
    assert(cd.next() == Some(3))
    assert(cd.next() == Some(2))
    assert(cd.next() == Some(1))
    assert(cd.next() == None)
}

test "CountDown next() after exhaustion is safe" {
    let mut cd = CountDown { current: 1 }
    cd.next()  // Some(1)
    assert(cd.next() == None)
    assert(cd.next() == None)  // повторный — не паникует
}

// for_loop.nv
test "for-in over CountDown sums correctly" {
    let mut sum = 0
    for n in CountDown { current: 4 } {
        sum = sum + n
    }
    assert(sum == 10)  // 4+3+2+1
}

// str_iters.nv
test "str.chars() iterates codepoints" {
    let mut chars []char = []char.new()
    for c in "Привет" {
        chars.push(c)
    }
    assert(chars.len() == 6)
    assert(chars[0] == 'П')
    assert(chars[5] == 'т')
}

test "str.bytes() iterates raw bytes" {
    let mut count = 0
    for _ in "Hi" {
        count = count + 1
    }
    assert(count == 2)
}

// generic_bound.nv
fn sum_iter[T Iter[int]](mut it T) -> int {
    let mut total = 0
    for n in it { total = total + n }
    total
}

test "generic sum_iter with CountDown" {
    assert(sum_iter(CountDown { current: 5 }) == 15)
}
```

## Acceptance criteria

- [ ] User-defined Iter — работает в for-in
- [ ] Exhaustion — None после конца, повторный next безопасен
- [ ] Пустой Iter — next() сразу None
- [ ] `str.chars()` / `str.bytes()` — работают как Iter в for-in
- [ ] Generic bound `Iter[T]` — компилируется и корректен
- [ ] Early break из for-in — корректен
- [ ] Compile-error для non-Iter в for-in
