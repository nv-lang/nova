##### `grammar.js` (minimal — verbatim baseline; productions expand в Ф.2/Ф.3)

```javascript
/**
 * @file Tree-sitter grammar for Nova
 * @author Evgeniy Golovin <unitcraft@nv-lang.org>
 * @license MIT
 */

/// <reference types="tree-sitter-cli/dsl" />
// @ts-check

const KEYWORDS = [
  // Source-of-truth: nv-lang/nova compiler-codegen/src/lexer/mod.rs:594-651
  'module', 'import', 'use', 'export', 'external',
  'fn', 'type', 'effect', 'alias',
  'let', 'const', 'mut', 'consume', 'readonly',
  'if', 'else', 'match', 'for', 'while', 'loop', 'in',
  'return', 'break', 'continue',
  'test',
  'true', 'false',
  'with', 'throw', 'as', 'is',
  'spawn', 'supervised', 'parallel', 'detach', 'blocking',
  'protocol', 'interrupt', 'forbid', 'realtime',
  'defer', 'errdefer',
  'select', 'lemma',
];

module.exports = grammar({
  name: 'nova',

  extras: $ => [
    /\s/,
    $.line_comment,
    $.doc_comment,
    $.block_comment,
  ],

  word: $ => $.identifier,

  rules: {
    // Top level — placeholder, expand в Ф.2
    source_file: $ => repeat($._top_level),

    _top_level: $ => choice(
      $._declaration_placeholder,
      $.identifier,
      $.number_literal,
      $.string_literal,
    ),

    _declaration_placeholder: $ => seq(
      choice(...KEYWORDS.map(k => alias(k, $.keyword))),
      optional($.identifier),
    ),

    // Lexical primitives — STABLE через Ф.2/Ф.3
    identifier: $ => /[a-zA-Z_][a-zA-Z0-9_]*/,

    number_literal: $ => token(choice(
      // Hex
      /0x[0-9a-fA-F][0-9a-fA-F_]*/,
      // Binary
      /0b[01][01_]*/,
      // Octal
      /0o[0-7][0-7_]*/,
      // Float
      /[0-9][0-9_]*\.[0-9][0-9_]*([eE][+-]?[0-9]+)?/,
      // Decimal int
      /[0-9][0-9_]*/,
    )),

    string_literal: $ => choice(
      seq('"', repeat(choice(
        /[^"\\$]/,
        /\\./,
        // Interpolation $...$  — refined в Ф.2 для nested expr
        seq('${', /[^}]*/, '}'),
      )), '"'),
    ),

    line_comment: $ => token(seq('//', /[^\r\n]*/)),
    doc_comment: $ => token(seq('///', /[^\r\n]*/)),
    block_comment: $ => token(seq('/*', /[^*]*\*+([^/*][^*]*\*+)*/, '/')),
  },
});
```

#### Productions добавляются в `grammar.js` (расширь, не переписывай)

**Expressions (Pratt-style precedence):**

```javascript
// Precedence levels (низкий → высокий)
const PREC = {
  assignment: 1,
  logical_or: 2,
  logical_and: 3,
  equality: 4,
  comparison: 5,
  bitwise_or: 6,
  bitwise_xor: 7,
  bitwise_and: 8,
  shift: 9,
  additive: 10,
  multiplicative: 11,
  unary: 12,
  call: 13,
  member: 14,
};

// В rules:
expression: $ => choice(
  $.literal_expr,
  $.path_expr,
  $.binary_expr,
  $.unary_expr,
  $.call_expr,
  $.method_call_expr,
  $.field_access_expr,
  $.index_expr,
  $.paren_expr,
  $.lambda_expr,
  $.match_expr,
  $.if_expr,
  $.block_expr,
  // ... etc
),

binary_expr: $ => choice(
  ...[
    ['||', PREC.logical_or],
    ['&&', PREC.logical_and],
    ['==', PREC.equality], ['!=', PREC.equality],
    ['<', PREC.comparison], ['<=', PREC.comparison],
    ['>', PREC.comparison], ['>=', PREC.comparison],
    ['+', PREC.additive], ['-', PREC.additive],
    ['*', PREC.multiplicative], ['/', PREC.multiplicative], ['%', PREC.multiplicative],
    // ... остальные операторы из spec/decisions/03-syntax.md
  ].map(([op, prec]) => prec.left(prec, seq(
    field('left', $.expression),
    field('op', op),
    field('right', $.expression),
  ))),
),
```

**Types:**

```javascript
type_expr: $ => choice(
  $.type_identifier,                                  // int, str, MyType
  $.generic_type,                                     // List[int], Map[K, V]
  $.tuple_type,                                       // (int, str)
  $.function_type,                                    // fn(int) -> str
  $.reference_type,                                   // &T, &mut T
),
```

**Declarations:**

```javascript
function_declaration: $ => seq(
  optional('export'),
  optional('external'),
  'fn',
  optional($.generic_params),                         // [T]
  field('name', $.identifier),
  $.parameter_list,
  optional(seq('->', field('return_type', $.type_expr))),
  optional($.where_clause),
  repeat($.contract_clause),                          // requires/ensures
  optional($.block_expr),
),

type_declaration: $ => seq(
  optional('export'),
  'type',
  field('name', $.type_identifier),
  optional($.generic_params),
  choice(
    $.sum_type_body,                                  // | Variant(...) | Variant
    $.record_type_body,                               // { field type }
    $.alias_type_body,                                // = T
  ),
),
```

#### Conflict resolution

Tree-sitter жалуется на conflicts → добавь `conflicts: $ => [...]` в grammar:

```javascript
conflicts: $ => [
  [$.expression, $.pattern],                          // expr vs match arm pattern
  [$.block_expr, $.record_type_body],                 // { ... } ambiguity
  // и т.д. — добавляй по ошибкам tree-sitter generate
],
```

#### Когда escalate к Opus 4.7

- Conflict resolution >5 итераций — Opus справится с более complex precedence math.
- Ambiguity между expression vs statement context (Nova не имеет explicit statement terminators) — design decision, не mechanical.
