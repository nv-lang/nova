## Ключевые тесты — round-trip

```nova
test "i64 BE round-trip" {
    let wb = WriteBuffer.new()
    wb.write_i64_be(-1_000_000)
    let rb = ReadBuffer.from(wb.into())
    assert(rb.read_i64_be() == -1_000_000)
}

test "f64 LE round-trip" {
    let wb = WriteBuffer.new()
    wb.write_f64_le(3.14159)
    let rb = ReadBuffer.from(wb.into())
    assert(rb.read_f64_le() == 3.14159)
}

test "char round-trip" {
    let wb = WriteBuffer.new()
    wb.write_char('Я')
    let rb = ReadBuffer.from(wb.into())
    assert(rb.read_char() == 'Я')
}
```
